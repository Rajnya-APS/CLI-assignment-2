#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <string.h>
#include <stdlib.h>

struct Employee {
    int id;
    char name[30];
    float salary;
};

int main() {

    struct Employee emp[2] = {
        {101, "Rahul", 30000},
        {102, "Amit", 35000}
    };

    int fd = open("employee.dat", O_RDWR | O_CREAT | O_TRUNC, 0644);

    if(fd < 0){
        perror("File Open Error");
        return 1;
    }

    write(fd, emp, sizeof(emp));

    printf("Employee records written successfully.\n");

    lseek(fd, sizeof(struct Employee), SEEK_SET);

    struct Employee update;

    read(fd, &update, sizeof(update));

    update.salary = 45000;

    lseek(fd, sizeof(struct Employee), SEEK_SET);

    write(fd, &update, sizeof(update));

    printf("Salary updated successfully.\n");

    lseek(fd, 0, SEEK_SET);

    struct Employee temp;

    printf("\nEmployee Records\n");
    printf("--------------------------\n");

    while(read(fd, &temp, sizeof(temp)) > 0){

        printf("ID : %d\n", temp.id);
        printf("Name : %s\n", temp.name);
        printf("Salary : %.2f\n\n", temp.salary);

    }

    close(fd);

    return 0;
}
